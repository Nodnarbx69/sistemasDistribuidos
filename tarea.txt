SELECT NOMBRE_PROV, DIRECCION_PROV, CIUDAD_PROV, EMAIL_PROV, NOMBRE_PRO, 
PRECIOVENTA_PRO 
FROM PROVEEDORES 
INNER JOIN PRODUCTOS ON CODIGO_PROV=FK_CODIGO_PROV 
WHERE NOMBRE_PROV LIKE '%S' AND CIUDAD_PROV='EL CARMEN' AND EMAIL_PROV LIKE '%HOTMAIL%';


SELECT CEDULA_CAJ, NOMBRE_CAJ, APELLIDO_CAJ, FECHAINGRESO_CAJ, NOMBRE_CLI, APELLIDO_CLI, FECHA_TRA, TIPO_TRA, TOTAL_TRA
FROM CAJEROS
INNER JOIN TRANSACCIONES ON CODIGO_CAJ=FK_CODIGO_CAJ
INNER JOIN CLIENTES ON CODIGO_CLI=FK_CODIGO_CLI
WHERE CEDULA_CAJ LIKE '13%' AND YEAR(FECHAINGRESO_CAJ)=2013 AND
LENGTH(APELLIDO_CAJ)=9;

SELECT CEDULA_CLI AS 'CEDULA',CONCAT(NOMBRE_CLI,' ',APELLIDO_CLI) AS
'CLIENTE',LOWER(EMAIL_CLI) AS 'EMAIL'
FROM CLIENTES
WHERE NOMBRE_CLI LIKE '%NA' AND APELLIDO_CLI LIKE '%L%';

SELECT FECHA_TRA, TIPO_TRA, TOTAL_TRA, CANTIDAD_DET, SUBTOTAL_DET, NOMBRE_PRO
FROM TRANSACCIONES
INNER JOIN DETALLES ON CODIGO_TRA=FK_CODIGO_TRA
INNER JOIN PRODUCTOS ON CODIGO_PRO=FK_CODIGO_PRO
WHERE TIPO_TRA='VENTA' AND TOTAL_TRA>=100 AND TOTAL_TRA<=500 AND
MONTH(FECHA_TRA)=12;
